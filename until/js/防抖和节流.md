## js工具
常用js工具类函数整理
### 防抖和节流
* debounce：防抖
```
/**
* debounce
* @desc 防抖
* @param immediate：立即执行
* @param wait：等待时间
* @param fn：执行的回调
* @return {String}
*/
debounce: function (immediate, wait, fn) {
  let timer = null
  return function () {
    let args = arguments
    let _this = this
    if (immediate && !timer) {
      fn.apply(_this, args)
    }
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(_this, args)
    }, wait)
  }
}
```
* throttle：节流
```
/**
* throttle
* @desc 节流
* @param wait：等待时间
* @param fn：执行的回调
* @return {String}
*/
throttle: function (immediate, wait, fn) {
  let timer = null
  let callNow = immediate
  return function () {
    let _this = this,
      args = arguments
    if (callNow) {
      fn.apply(_this, args)
      callNow = false
    }
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(_this, args)
        timer = null
      }, wait)
    }
  }
}
```
